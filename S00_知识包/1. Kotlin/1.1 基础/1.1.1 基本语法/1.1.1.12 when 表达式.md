### 🔀 when 表达式

```kotlin
fun describe(obj: Any): String =
    when (obj) {
        1          -> "一"
        "Hello"    -> "Hello"
        is Long    -> "是长整型"
        !is String -> "不是字符串"
        else       -> "未知"
    }

fun main() {
    println(describe(1))
    println(describe("Hello"))
    println(describe(1000L))
    println(describe(2))
    println(describe("other"))
}

// 一
// Hello
// 是长整型
// 不是字符串
// 未知
```
