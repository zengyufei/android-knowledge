### 🔍 类型检测与自动类型转换

#### 🌟 `is` 操作符用于检测一个表达式是否为某类型的实例。

#### 🔮 如果一个不可变的局部变量或属性已被判断为某类型，那么在检测后的分支中可以直接将其视为该类型使用，无需显式转换：

```kotlin
//sampleStart
fun getStringLength(obj: Any): Int? {
    if (obj is String) {
        // `obj` 在该条件分支内自动转换成 `String`
        return obj.length
    }

    // 在离开类型检测分支后，`obj` 仍然是 `Any` 类型
    return null
}
//sampleEnd

fun main() {
    fun printLength(obj: Any) {
        println("获取'$obj'的长度。结果：${getStringLength(obj) ?: "错误：该对象不是字符串"} ")
    }
    printLength("不可理喻")
    printLength(1000)
    printLength(listOf(Any()))
}

// 获取'不可理喻'的长度。结果：4 
// 获取'1000'的长度。结果：错误：该对象不是字符串 
// 获取'[java.lang.Object@50040f0c]'的长度。结果：错误：该对象不是字符串 
```

#### 🔀 或者：

```kotlin
//sampleStart
fun getStringLength(obj: Any): Int? {
    if (obj !is String) return null

    // `obj` 在这一分支自动转换为 `String`
    return obj.length
}
//sampleEnd

fun main() {
    fun printLength(obj: Any) {
        println("获取'$obj'的长度。结果：${getStringLength(obj) ?: "错误：该对象不是字符串"} ")
    }
    printLength("不可理喻")
    printLength(1000)
    printLength(listOf(Any()))
}

// 获取'不可理喻'的长度。结果：4 
// 获取'1000'的长度。结果：错误：该对象不是字符串 
// 获取'[java.lang.Object@50040f0c]'的长度。结果：错误：该对象不是字符串 
```
#### 🔄 甚至：

```kotlin
//sampleStart
fun getStringLength(obj: Any): Int? {
    // `obj` 在 `&&` 右边自动转换成 `String` 类型
    if (obj is String && obj.length > 0) {
        return obj.length
    }

    return null
}
//sampleEnd

fun main() {
    fun printLength(obj: Any) {
        println("获取'$obj'的长度。结果：${getStringLength(obj) ?: "错误：该对象不是字符串"} ")
    }
    printLength("不可理喻")
    printLength("")
    printLength(1000)
}

// 获取'不可理喻'的长度。结果：4 
// 获取''的长度。结果：错误：该对象不是字符串 
// 获取'1000'的长度。结果：错误：该对象不是字符串 
```
